on_actions = {
	on_startup = {
		effect = {
			every_country = {
                set_variable = { money = 30 }
                set_variable = { money_rate = 5 }
                set_variable = { money_max = 10000000 }
                clamp_variable = { var = money min = 0 max = money_max }
            }
		}
	}
    
    on_daily = {
		effect = {
			add_to_variable = { money = money_rate }
        }
    }
}
