		visible = {
            has_country_leader = { ruling_only = yes name = "Winston Churchill" }

				set_variable = { selector = 3 }

					check_variable = { selector = 2 }
				}
			}
			defensive_buildings_click_enabled = {
				NOT ={
					check_variable = { selector = 3 }
				}
			}
		}
	}
